  unsafe {
    println!(

    );

  // Многопоточное чтение и запись

}

pub struct Config {
  path: PathBuf,

  flag: flag::ENV,
  sync_period: u64,
  sync_bytes: u64,

lazy_static! {
  pub static ref ENV_CONFIG_DEFAULT: Config = Config {
    path:PathBuf::new(),
    mode: 0o600,
    //https://github.com/erthink/libmdbx/issues/248
    sync_period : 65536, // За 1/65536 секунды

      | flag::ENV::MDBX_COALESCE

  };
}

# Флаги базы данных

Вы можете увидеть флаги базы данных, добавленные к данным в файле examples/02.rsОткрыть в новом окне по адресу Test4. flag DUPSORT

База данных libmdbx имеет ряд флагов ( MDBX_db_flags_tОткрыть в новом окне ), которые могут быть установлены.

  • REVERSEKEY использует обратное сравнение строк для ключей. (полезно при использовании небольших конечных кодовых чисел в качестве ключей)
  • DUPSORT использует отсортированные дубликаты, т.е. допускает несколько значений для ключа.
  • INTEGERKEY Родной упорядоченный по байтам числовой ключ uint32_t или uint64_t. Ключи должны быть одинакового размера и должны быть выровнены при передаче в качестве аргументов.
  • DUPFIXED Размер значений данных должен быть одинаковым, если используется DUPSORT (позволяет быстро подсчитать количество значений).
  • DUPSORT и DUPFIXED требуются для INTEGERDUP; значения являются целыми числами (аналогично INTEGERKEY). Все значения данных должны иметь одинаковый размер и должны быть выровнены при передаче в качестве параметров.
  • REVERSEDUP использует DUPSORT; для значений данных используется обратное сравнение строк.
  • CREATE создает БД, если она не существует (добавляется по умолчанию).
  • DB_ACCEDE Открывает существующую подбазу данных, созданную с использованием флага неизвестности.
    Этот флаг DB_ACCEDE предназначен для открытия существующих подбаз данных, созданных с неизвестными флагами (REVERSEKEY, DUPSORT, INTEGERKEY, DUPFIXED, INTEGERDUP и REVERSEDUP).
    В этом случае подбаза данных не вернет ошибку INCOMPATIBLE, а будет открыта с флагами, использованными для ее создания, и приложение сможет определить фактические флаги с помощью mdbx_dbi_flags().
# DUPSORT : Один ключ соответствует более чем одному значению

DUPSORTЭто означает, что ключ может соответствовать более чем одному значению.

Если вы хотите установить несколько флагов, напишите следующим образом flag DUPSORT | DUPFIXED

# .dup(key) итератор, возвращающий все значения, соответствующие ключу

Эта функция доступна только для баз данных, помеченных DUPSORT, где ключ может соответствовать более чем одному значению.

Для баз данных DUPSORT, get возвращает только первое значение для этого ключа. Чтобы получить все значения, используйте dup.

# Флаги базы данных, автоматически добавляемые по умолчанию

Если тип данных u32 / u64 / usize, автоматически добавляется флаг базы данных INTEGERKEYОткрыть в новом окне .

На машинах с кодировкой малого конца автоматически добавляются другие типы цифр REVERSEKEYОткрыть в новом окне Флаг базы данных автоматически добавляется, если тип данных - / / .

# Удаление данных

# .del(key) Удаление клавиши

.del(val) Удаляет значение, соответствующее ключу.

Если база данных имеет флаг DUPSORT, все значения под этим ключом будут удалены.

Возвращает true, если данные удалены, и false, если нет.

# .del_val(key,val) Удаление точного совпадения

.del_val(key,val) Удаляет пары ключ-значение, которые точно соответствуют входным параметрам.

Возвращает true, если данные удалены, и false, если нет.

# Сортировка

Ключи libmdbx сортируются в порядке словаряОткрыть в новом окне.

  • Для беззнаковых чисел

    сортируются от наименьшего к наибольшему, поскольку флаги базы данных добавляются автоматически ( u32/ u64/ usize добавляются к INTEGERKEY, другие добавляются к REVERSEKEY в зависимости от машинного кода).

  • Для знаковых чисел

    порядок следующий: сначала 0, затем все положительные числа от наименьшего до наибольшего, затем все отрицательные числа от наименьшего до наибольшего.

# .range(begin..end) Интервал Итерация

Для чисел интервал - это числовой интервал.

Для двоичного числа можно построить такой же интервал, например

let begin : &[u8] = &[1,1];
for (k,v) in test0.range(begin..=&[2]) {}

# .rev_range Инвертированные интервалы

Если вы хотите получить инвертированный интервал, который меньше или равен значению, вы можете сделать следующее

test2.rev_range(2..)

Поместите выход

(2, 4)
(1, 5)
(0, 0)

Один из begin или end не должен быть установлен для инвертированного интервала; потому что если оба установлены, вы всегда можете использовать range(end..begin) для достижения того же эффекта.

# Сноски

Они приводят преимущества перехода с LMDB на MDBX.

Erigon начинал с бэкенда базы данных BoltDB, затем добавил поддержку BadgerDB, и, наконец, полностью перешел на LMDB. В какой-то момент мы столкнулись с проблемами стабильности, вызванными использованием LMDB, которые не были предусмотрены создателями. С тех пор мы смотрим на хорошо поддерживаемую производную LMDB под названием MDBX и надеемся использовать их улучшения стабильности и потенциально сотрудничать с ними в будущем. Интеграция MDBX теперь завершена, и пришло время для дальнейшего тестирования и документирования.

Преимущества перехода с LMDB на MDBX.

  1. Рост "пространства (геометрии)" файлов базы данных работает правильно. Это важно, особенно в Windows. В LMDB необходимо один раз заранее указать размер карты памяти (в настоящее время по умолчанию мы используем 2 Тб), и если файл базы данных вырастет сверх этого предела, процесс необходимо перезапустить. В Windows установка размера карты памяти в 2 Тб приведет к тому, что файл базы данных изначально будет иметь размер 2 Тб, что не очень удобно. В MDBX размер карты памяти увеличивается с шагом в 2 Гб. Это означает периодическое переключение, но в результате повышается удобство использования.

  2. MDBX имеет более строгие проверки на одновременное использование обработки транзакций и перекрытие транзакций чтения и записи в одном потоке выполнения. Это позволяет нам выявить некоторые неочевидные ошибки и делает поведение более предсказуемым.
    За более чем 5 лет (с момента отделения от LMDB) MDBX накопил большое количество исправлений безопасности и heisenbug, которые, насколько нам известно, все еще существуют в LMDB. Некоторые из них были обнаружены во время нашего тестирования, и сопровождающие MDBX отнеслись к ним серьезно и быстро исправили их.

  3. Когда речь идет о базах данных, которые постоянно модифицируют данные, они создают значительное количество восстанавливаемого пространства (также известного как "freelist" в терминологии LMDB). Нам пришлось внести исправления в LMDB, чтобы исправить наиболее серьезные недостатки в работе с восстанавливаемым пространством (анализ)Открыть в новом окне. MDBX уделил особое внимание эффективной работе с восстанавливаемым пространством и до сих пор не нуждался в исправленияхОткрыть в новом окне.

  4. По результатам нашего тестирования, MDBX показал себя немного лучше на наших рабочих нагрузках.

  5. MDBX раскрывает больше данных внутренней телеметрии - больше метрик о том, что происходит внутри базы данных. И у нас есть эти данные в Grafana - для принятия лучших решений по проектированию приложений. Например, после полного перехода на MDBX (удаление поддержки LMDB), мы внедрим политику "фиксации наполовину полной транзакции", чтобы избежать переполнения/недополнения дисковых контактов. Это еще больше упростит наш код без ущерба для производительности.

  6. MDBX поддерживает режим "Exclusive open" - мы используем его для миграции баз данных, чтобы предотвратить доступ к базе данных других читателей во время процесса миграции.
