# Quickjsを改造してRust関数をインポートする - Krakenを拡張する新しい考え方

# コード保管庫

  • ギブス新しいウィンドウで開く
  • ゲテモノ新しいウィンドウで開く
  • ギトフリック新しいウィンドウで開く
  • ビットバケット新しいウィンドウで開く

# 創世記

Krakenは新しいウィンドウで開く 、 Flutter 、スクリプトエンジンとして quickjsを新しいウィンドウで開く使用した、高性能なWebレンダリングエンジンです。

rust を使って Kraken の拡張をいくつか書きたいと思いました。

Krakenは、 dart新しいウィンドウで開く を使用した拡張機能の書き込みをサポートしています。

使用方法 flutter_rust_bridge新しいウィンドウで開く rust 、 dart。

この2点を組み合わせれば、 rust を使って Kraken の拡張を書くことは難しくありません。
しかし、このソリューションでは、 dart が rust を呼び出す際のパフォーマンスペナルティと、quickjs が dart を呼び出す際のパフォーマンスペナルティがあるため、パフォーマンスのオーバーヘッドが大きく感じられます。

一方、 rust のコミュニティでは rquickjs新しいウィンドウで開く このような呼び出しは、 rust の quickjs ライブラリに対して行われます。
しかし、それらは quickjsを埋め込む代わりに quickjs を呼び出し、 quickjs のマジックには使えません。

このコードベースでは、 quickjs のソースコードを直接修正して、 rust 拡張に対応させるという新しい解決策を実装しています。

これは、Krakenの改造だけでなく、 quickjs を埋め込むすべてのフレームワークとライブラリに使用できる汎用的なソリューションです。

# デモの様子

test.jsのコードは以下の通りです。

const try_run = (func, ...args) => {
  try {
    func(...args)
  } catch (err) {
    console.log('❌', err.message)
    console.log(err.stack)
  }
};

import * as rust from 'rust'
console.log("export from rust :", Object.keys(rust))

import {

(async () => {

# rustでのfibの実装

jsでインポートしたfib関数は、 rust/src/export/fib.rs 、コードは以下の通りです。

use crate::js::{self, arg};
use quickjs_ffi::{JSContext, JSValue};
use rust_macro::js;
use std::os::raw::c_int;

#[js]
pub fn fib(n: i64) -> i64 {
  if n <= 1 {
    return if n == 1 { 1 } else { 0 };
  }
  n + fib(n - 1)
}

#[no_mangle]
pub extern "C" fn js_fib(
  ctx: *mut JSContext,
  _this: JSValue,
  argc: c_int,
  argv: *mut JSValue,
) -> JSValue {
  if let Err(err) = arg::arg_miss(ctx, argc, fib_args_len) {
    return err;
  }
  match arg::arg_i64(ctx, argv, 0) {
    Err(err) => err,
    Ok(n) => js::val(ctx, fib(n)),
  }
}

現在、プロシージャマクロ #[js] は、関数への引数の数を特定する定数 fib_args_lenを追加しているだけです。

将来的には、プロシージャマクロ ./rust_macro を記述することで、完全に自動で機能を書き出すことができるようになります。

use crate::{js::arg, qjs::run};
use async_io::Timer;
use quickjs_ffi::{JSContext, JSValue};
use rust_macro::js;
use std::{os::raw::c_int, time::Duration};

#[js]
pub async fn sleep(n: u64) {
  Timer::after(Duration::from_millis(n)).await;
}

#[no_mangle]
pub extern "C" fn js_sleep(
  ctx: *mut JSContext,
  _this: JSValue,
  argc: c_int,
  argv: *mut JSValue,
) -> JSValue {
  if let Err(err) = arg::arg_miss(ctx, argc, sleep_args_len) {
    return err;
  }
  match arg::arg_i64(ctx, argv, 0) {
    Err(err) => err,
    Ok(n) => run(ctx, async move {
      sleep(n as u64).await;
      Ok(())
    }),
  }
}

上で見たように、エクスポートされた関数はすべてディレクトリ ./rust/src/export で定義されています。このディレクトリ mod.rs は ./rust/build.xsh を実行すると自動的に生成され、その下にすべての .rs ファイルがエクスポートされます。

pub(crate) fn arg_i64(ctx: *mut JSContext, argv: *mut JSValue, pos: isize) -> Result<i64, JSValue> {
  unsafe {
    let val = *argv.offset(pos);
    if JS_IsNumber(val) == 0 {
      throw(ctx, format!("not number : args position {pos}"))?
    }
    let mut n = MaybeUninit::uninit();
    JS_ToInt64(ctx, n.as_mut_ptr() as _, val);
    Ok(n.assume_init())
  }
}

現在は、引数の数だけチェックし、i64型を読み込んでいます。

use quickjs_ffi::{JSContext, JSValue, JS_NewInt64, JS_NewString, JS_NULL, JS_UNDEFINED};
use std::ffi::CString;

pub enum Val {
  None,
  Undefined,
  I64(i64),
  CString(CString),
}

impl From<()> for Val {
  fn from(_: ()) -> Self {
    Val::Undefined
  }
}

impl From<i64> for Val {
  fn from(t: i64) -> Self {
    Val::I64(t)
  }
}

impl From<CString> for Val {
  fn from(t: CString) -> Self {
    Val::CString(t)
  }
}

pub(crate) fn val(ctx: *mut JSContext, t: impl Into<Val>) -> JSValue {
  match t.into() {
    Val::None => JS_NULL,
    Val::Undefined => JS_UNDEFINED,
    Val::I64(n) => unsafe { JS_NewInt64(ctx, n) },
    Val::CString(cstr) => unsafe { JS_NewString(ctx, cstr.as_ptr()) },
  }
}

None,(),i64, CString から js への変換は4種類のみ定義されています。好きなだけ追加できます。

その他のデータ型は、 JS_New で始まる qjs_sys新しいウィンドウで開く の関数で宣言することができる。

# 原理の説明

# quickjs_rust/patch.py

./quickjs_rust/patch.py を実行すると、 quickjs のソースコードに若干の変更が加えられます。

関数 JS_AddRust の一つで、rust モジュールに注入するために使用されます。

rust_run を JS_ExecutePendingJob にインジェクションして非同期関数を呼び出す。

すべての変更点のスクリーンショットは以下の通りです。

# quickjs_rust.h

上記の変更点から、新しいヘッダーファイル quickjs_rust.h を導入し、以下のようなコードを記述していることがわかります。

#ifndef QUICKJS_RUST_H
#define QUICKJS_RUST_H

#include "../quickjs/quickjs.h"
#include "../rust/rust.h"

#define countof(x) (sizeof(x) / sizeof((x)[0]))
#define JS_RUSTFUNC_DEF(name) JS_CFUNC_DEF(#name, name##_args_len, js_##name)
#include "./js_rust_funcs.h"

static const unsigned int js_rust_funcs_count = countof(js_rust_funcs);

static int
js_rust_init(JSContext* ctx, JSModuleDef* m)
{
  return JS_SetModuleExportList(ctx, m, js_rust_funcs,
      js_rust_funcs_count);
}

#define JS_INIT_MODULE js_init_module_rust

JSModuleDef* JS_INIT_MODULE(JSContext* ctx, const char* module_name)
{
  JSModuleDef* m;
  m = JS_NewCModule(ctx, module_name, js_rust_init);
  if (!m)
    return NULL;
  js_rust_init(ctx, m);
  return m;
}

void JS_AddRust(JSContext* ctx, JSRuntime* rt)
{
  JSModuleDef* m = JS_INIT_MODULE(ctx, "rust");
  for (unsigned int i = 0; i < js_rust_funcs_count; i++) {
    JS_AddModuleExport(ctx, m, js_rust_funcs[i].name);
  }
  rust_init(ctx, rt);
}

#endif

# rust/rust.h

quickjs_rust/quickjs_rust.h が quickjs_rust/js_rust_funcs.hを紹介しているのがわかりますが、これは rust export 関数のヘッダーファイル rust/rust.h から自動的に生成されるもので、手で修正してはいけません。

そして、 ./rust/build.xsh から cbindgen を呼び出すことで rust/rust.h が生成される。

# rust/build.xsh

#!/usr/bin/env xonsh

from os.path import dirname,abspath
import platform
PWD = dirname(abspath(__file__))
cd @(PWD)

p"../.xonshrc".exists() && source ../.xonshrc

./src/export/mod.gen.py

system = platform.system().lower()
if system == 'darwin':
  system = f'apple-{system}'

TARGET=f'{platform.machine()}-{system}'

def cbindgen():
  cbindgen -q --config cbindgen.toml --crate rust --output rust.h

try:
  cbindgen()
except:
  cargo clean
  cbindgen()

cargo build \
--release \
-Z build-std=std,panic_abort \
-Z build-std-features=panic_immediate_abort \
--target @(TARGET)

mv ./target/@(TARGET)/release/librust.a ./target/release

pub(crate) fn arg_i64(ctx: *mut JSContext, argv: *mut JSValue, pos: isize) -> Result<i64, JSValue> {
  unsafe {
    let val = *argv.offset(pos);
    if JS_IsNumber(val) == 0 {
      throw(ctx, format!("not number : args position {pos}"))?
    }
    let mut n = MaybeUninit::uninit();
    JS_ToInt64(ctx, n.as_mut_ptr() as _, val);
    Ok(n.assume_init())
  }
